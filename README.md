def find_duplicates(nums):
    seen = set()
    duplicates = set()
    for num in nums:
        if num in seen:
            duplicates.add(num)
        else:
            seen.add(num)
    return list(duplicates)

numbers = [1, 2, 3, 4, 5, 1, 2, 3, 6, 7, 8, 9]

print(find_duplicates(numbers))
